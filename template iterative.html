import matplotlib.pyplot as plt

def dibujarTray(tray1, tray2):
    xs1 = [float(vec[0]) for vec in tray1]
    ys1 = [float(vec[1]) for vec in tray1]

    xs2 = [float(vec[0]) for vec in tray2]
    ys2 = [float(vec[1]) for vec in tray2]

    plt.plot(xs1, ys1, marker="o", label="Gauss Jacobi")
    plt.plot(xs2, ys2, marker="o", label="Gauss Seidel")
    plt.title("Trayectorias")
    plt.xlabel("X")
    plt.ylabel("Y")
    plt.grid(True)
    plt.legend()
    plt.show()

    plt.ylabel("Y")
    plt.grid(True)
    plt.legend()
